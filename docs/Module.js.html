const hashString = function (string) {
    let hash = 2166136261 >>> 0;
    let index = 0;
    while (index < string.length) {
        hash ^= string.charCodeAt(index);
        hash = Math.imul(hash, 16777619) >>> 0;
        index += 1;
    }
    return hash >>> 0;
};

const seedToState = (seed) => (
    typeof seed === "string"
    ? hashString(seed)
    : seed >>> 0
);

const nextRandom = function (rngState) {
    const next = (
        Math.imul(1664525, rngState >>> 0) + 1013904223
    ) >>> 0;
    return {rngState: next, value: next / 4294967296};
};

const randomInt = function (rngState, minimum, maximumExclusive) {
    if (
        !Number.isInteger(minimum) ||
        !Number.isInteger(maximumExclusive) ||
        maximumExclusive <= minimum
    ) {
        throw new RangeError("Invalid random integer interval.");
    }
    const result = nextRandom(rngState);
    return {
        rngState: result.rngState,
        value: minimum + Math.floor(
            result.value * (maximumExclusive - minimum)
        )
    };
};

/**
 * Return a deterministically shuffled copy of a list and the next RNG state.
 * @memberof LastCurtain
 * @template T
 * @param {number} rngState Current unsigned 32-bit random state.
 * @param {T[]} items Items to shuffle.
 * @returns {{rngState:number, shuffled:T[]}} Shuffled copy and next state.
 */
export function shuffleWithSeed(rngState, items) {
    const shuffled = [...items];
    let state = rngState;
    let index = shuffled.length - 1;
    while (index >= 1) {
        const result = randomInt(state, 0, index + 1);
        const otherIndex = result.value;
        const temporary = shuffled[index];
        state = result.rngState;
        shuffled[index] = shuffled[otherIndex];
        shuffled[otherIndex] = temporary;
        index -= 1;
    }
    return {rngState: state, shuffled};
}

/**
 * Create the standard ordered 52-card theatre deck.
 * @memberof LastCurtain
 * @returns {Card[]} New ordered deck.
 */
export function createDeck() {
    return R.chain(function (suit) {
        return R.map(function (rank) {
            return {
                id: `${suit}-${rank}`,
                suit,
                rank,
                color: (
                    suit === "H" || suit === "D"
                    ? "red"
                    : "black"
                ),
                value: (
                    rank === 1
                    ? 11
                    : rank >= 11
                    ? 10
                    : rank
                )
            };
        }, RANKS);
    }, SUITS);
}

const rankStrength = (rank) => (rank === 1 ? 14 : rank);
const cardIdsForRanks = (cards, ranks) => cards.filter(
    (card) => ranks.includes(card.rank)
).map((card) => card.id);

/**
 * Recognise the highest natural poker hand in one to five selected cards.
 * @memberof LastCurtain
 * @param {Card[]} cards Cards forming a proposed scene.
 * @returns {Evaluation} Natural hand evaluation.
 */
export function classifyNaturalHand(cards) {
    if (!Array.isArray(cards) || cards.length < 1 || cards.length > 5) {
        throw new RangeError("A hand must contain one to five cards.");
    }
    const rankGroups = R.countBy((card) => String(card.rank), cards);
    const groupedRanks = Object.keys(rankGroups).map(Number).sort(
        (left, right) => (
            rankGroups[String(right)] - rankGroups[String(left)] ||
            rankStrength(right) - rankStrength(left)
        )
    );
    const counts = groupedRanks.map((rank) => rankGroups[String(rank)]);
    const uniqueRanks = cards.reduce(
        (ranks, card) => addUnique(ranks, card.rank),
        []
    );
    const highRanks = uniqueRanks.map(rankStrength).sort((a, b) => a - b);
    const aceLow = (
        uniqueRanks.length === 5 &&
        [1, 2, 3, 4, 5].every((rank) => uniqueRanks.includes(rank))
    );
    const consecutive = (
        highRanks.length === 5 &&
        highRanks.every((rank, index) => (
            index === 0 || rank === highRanks[index - 1] + 1
        ))
    );
    const straight = cards.length === 5 && (aceLow || consecutive);
    const flush = cards.length === 5 && cards.every(
        (card) => card.suit === cards[0].suit
    );
    let type = "HighCard";
    let contributingRanks = [];
    if (straight && flush) {
        type = "StraightFlush";
        contributingRanks = uniqueRanks;
    } else if (counts[0] === 4) {
        type = "FourOfAKind";
        contributingRanks = [groupedRanks[0]];
    } else if (
        cards.length === 5 &&
        counts[0] === 3 &&
        counts[1] === 2
    ) {
        type = "FullHouse";
        contributingRanks = groupedRanks.slice(0, 2);
    } else if (flush) {
        type = "Flush";
        contributingRanks = uniqueRanks;
    } else if (straight) {
        type = "Straight";
        contributingRanks = uniqueRanks;
    } else if (counts[0] === 3) {
        type = "ThreeOfAKind";
        contributingRanks = [groupedRanks[0]];
    } else if (counts[0] === 2 && counts[1] === 2) {
        type = "TwoPair";
        contributingRanks = groupedRanks.slice(0, 2);
    } else if (counts[0] === 2) {
        type = "Pair";
        contributingRanks = [groupedRanks[0]];
    } else {
        const highest = cards.reduce(function (best, card) {
            return rankStrength(card.rank) > rankStrength(best.rank)
                ? card
                : best;
        }, cards[0]);
        contributingRanks = [highest.rank];
    }
    const contributingCardIds = cardIdsForRanks(cards, contributingRanks);
    return {
        type,
        contributingCardIds,
        kickerCardIds: cards.filter(
            (card) => !contributingCardIds.includes(card.id)
        ).map((card) => card.id),
        priority: HAND_PRIORITY[type],
        modifiedByRoleId: null,
        modificationLabel: null
    };
}

const duplicateItems = (items) => items.some(
    (item, index) => items.indexOf(item) !== index
);

const selectionError = function (game) {
    if (game.phase !== "playing") {
        return "WRONG_PHASE";
    }
    if (game.selectedCardIds.length === 0) {
        return "NO_CARDS_SELECTED";
    }
    if (game.selectedCardIds.length > 5) {
        return "TOO_MANY_CARDS";
    }
    if (duplicateItems(game.selectedCardIds)) {
        return "DUPLICATE_CARD_SELECTION";
    }
    if (game.selectedCardIds.some(
        (id) => !game.hand.some((card) => card.id === id)
    )) {
        return "CARD_NOT_IN_HAND";
    }
    return null;
};

const selectedCards = (game) => game.selectedCardIds.map(
    (id) => game.hand.find((card) => card.id === id)
);

const addTrigger = (ids, roleId) => addUnique(ids, roleId);

const applyTwinMask = function (game, cards, natural, triggeredRoleIds, effectLog) {
    if (!game.castSlots.includes("twinMask") || cards.length !== 5) {
        return {evaluation: natural, triggeredRoleIds, effectLog};
    }
    const face = game.roleState.twinMask.face;
    const requiredColor = face === "comedy" ? "red" : "black";
    if (!cards.every((card) => card.color === requiredColor)) {
        return {evaluation: natural, triggeredRoleIds, effectLog};
    }
    const virtualCards = cards.map((card) => ({
        ...card,
        suit: requiredColor
    }));
    const upgraded = classifyNaturalHand(virtualCards);
    if (upgraded.priority <= natural.priority) {
        return {evaluation: natural, triggeredRoleIds, effectLog};
    }
    const label = `via ${face === "comedy" ? "Comedy" : "Tragedy"} Mask`;
    return {
        evaluation: {
            ...upgraded,
            modifiedByRoleId: "twinMask",
            modificationLabel: label
        },
        triggeredRoleIds: addTrigger(triggeredRoleIds, "twinMask"),
        effectLog: [...effectLog, {
            roleId: "twinMask",
            message: `The Twin Mask elevated the scene to ${upgraded.type}.`
        }]
    };
};

const attemptRehearsalRestore = function (state) {
    const attemptedDelta = state.change.attemptedDelta + 1;
    if (state.rehearsals < 3) {
        return {
            rehearsals: state.rehearsals + 1,
            change: {
                ...state.change,
                attemptedDelta,
                appliedDelta: state.change.appliedDelta + 1,
                after: state.rehearsals + 1
            },
            restored: true
        };
    }
    return {
        rehearsals: state.rehearsals,
        change: {
            ...state.change,
            attemptedDelta,
            lostToCapacity: state.change.lostToCapacity + 1,
            after: state.rehearsals
        },
        restored: false
    };
};

/**
 * Calculate a selected scene without changing cards, state, or random state.
 * @param {GameState} game Current game.
 * @returns {SceneCalculation} Resolution and optional commit.
 */
const resolveScene = function (game) {
    const errorCode = selectionError(game);
    if (errorCode) {
        return {resolution: invalidResolution(game, errorCode), commit: null};
    }
    const cards = selectedCards(game);
    const naturalEvaluation = classifyNaturalHand(cards);
    let triggeredRoleIds = [];
    let effectLog = [];
    const twinResult = applyTwinMask(
        game,
        cards,
        naturalEvaluation,
        triggeredRoleIds,
        effectLog
    );
    const finalEvaluation = twinResult.evaluation;
    triggeredRoleIds = twinResult.triggeredRoleIds;
    effectLog = twinResult.effectLog;
    if (game.bannedHandTypes.includes(finalEvaluation.type)) {
        return {
            resolution: invalidResolution(
                game,
                "HAND_TYPE_FORBIDDEN",
                `${finalEvaluation.type} was banned by the retake.`
            ),
            commit: null
        };
    }

    const roleState = copyRoleState(game.roleState);
    const sameAsPrevious = (
        game.sceneNumber > 1 &&
        finalEvaluation.type === game.previousHandType
    );
    const facelessAbsorbed = (
        game.castSlots.includes("faceless") &&
        roleState.faceless.ready &&
        sameAsPrevious
    );
    if (facelessAbsorbed) {
        roleState.faceless.ready = false;
        triggeredRoleIds = addTrigger(triggeredRoleIds, "faceless");
        effectLog.push({
            roleId: "faceless",
            message: "The Faceless Understudy absorbed the repeated hand."
        });
    } else if (
        game.castSlots.includes("faceless") &&
        !roleState.faceless.ready &&
        finalEvaluation.type !== game.previousHandType
    ) {
        roleState.faceless.ready = true;
    }
    const repeated = sameAsPrevious && !facelessAbsorbed;
    const varietyBefore = game.varietyLevel;
    const varietyAfter = (
        game.sceneNumber === 1
        ? 0
        : repeated
        ? 0
        : Math.min(3, varietyBefore + 1)
    );
    const baseScore = repeated ? 0 : HAND_BASE[finalEvaluation.type];
    const cardValue = cards.reduce((sum, card) => sum + card.value, 0);
    const varietyScore = 15 * varietyAfter;
    const preRoleScore = baseScore + cardValue + varietyScore;
    const scoreItems = [];
    if (repeated) {
        scoreItems.push({id: "repeat", label: "REPEATED", amount: 0});
    }
    scoreItems.push({id: "base", label: "Base Scene", amount: baseScore});
    scoreItems.push({id: "cards", label: "Card Values", amount: cardValue});
    scoreItems.push({
        id: "variety",
        label: `Variety ${varietyAfter}`,
        amount: varietyScore
    });

    if (game.castSlots.includes("lastEntrant")) {
        if (game.sceneNumber === 1 || game.sceneNumber === 2) {
            const immediate = Math.floor(preRoleScore / 2);
            const stored = preRoleScore - immediate;
            roleState.lastEntrant.delayedApplause += stored;
            scoreItems.push({
                id: "stored",
                label: "Stored for Final Entrance",
                amount: -stored
            });
            triggeredRoleIds = addTrigger(triggeredRoleIds, "lastEntrant");
            effectLog.push({
                roleId: "lastEntrant",
                message: `Stored ${stored} delayed applause.`
            });
        } else if (game.sceneNumber === 5) {
            const finale = roleState.lastEntrant.delayedApplause * 2;
            scoreItems.push({
                id: "finale",
                label: "Delayed Applause x2",
                amount: finale
            });
            triggeredRoleIds = addTrigger(triggeredRoleIds, "lastEntrant");
            effectLog.push({
                roleId: "lastEntrant",
                message: `Released ${finale} delayed applause.`
            });
        }
    }

    let rehearsals = game.rehearsals;
    let witnessedHandTypes = [...game.witnessedHandTypes];
    let rehearsalChange = zeroResourceChanges(game).rehearsals;
    const witnessedAdded = [];
    game.castSlots.forEach(function (roleId) {
        if (
            roleId === "witnesses" &&
            !witnessedHandTypes.includes(finalEvaluation.type)
        ) {
            witnessedHandTypes = addUnique(
                witnessedHandTypes,
                finalEvaluation.type
            );
            witnessedAdded.push(finalEvaluation.type);
            triggeredRoleIds = addTrigger(triggeredRoleIds, "witnesses");
            effectLog.push({
                roleId: "witnesses",
                message: `Witnessed ${finalEvaluation.type} for the first time.`
            });
            if (witnessedHandTypes.length % 3 === 0) {
                const restored = attemptRehearsalRestore({
                    rehearsals,
                    change: rehearsalChange
                });
                rehearsals = restored.rehearsals;
                rehearsalChange = restored.change;
                effectLog.push({
                    roleId: "witnesses",
                    message: restored.restored
                        ? "Restored one rehearsal."
                        : "Rehearsal capacity full — restoration lost."
                });
            }
        }
        if (roleId === "curtainDirector") {
            const index = game.castSlots.indexOf("curtainDirector");
            const leftRoleId = index > 0 ? game.castSlots[index - 1] : null;
            if (leftRoleId && triggeredRoleIds.includes(leftRoleId)) {
                roleState.curtainDirector.cueTokens += 1;
                triggeredRoleIds = addTrigger(
                    triggeredRoleIds,
                    "curtainDirector"
                );
                effectLog.push({
                    roleId: "curtainDirector",
                    message: `Recorded a cue from ${ROLE_DEFINITIONS[leftRoleId].name}.`
                });
                if (roleState.curtainDirector.cueTokens >= 2) {
                    const restored = attemptRehearsalRestore({
                        rehearsals,
                        change: rehearsalChange
                    });
                    rehearsals = restored.rehearsals;
                    rehearsalChange = restored.change;
                    roleState.curtainDirector.cueTokens = 0;
                    effectLog.push({
                        roleId: "curtainDirector",
                        message: restored.restored
                            ? "Two cues restored one rehearsal."
                            : "Two cues resolved; restoration was lost at capacity."
                    });
                }
            }
        }
    });

    const totalScore = scoreItems.reduce((sum, item) => sum + item.amount, 0);
    const resourceChanges = {
        rehearsals: rehearsalChange,
        delayedApplause: {
            before: game.roleState.lastEntrant.delayedApplause,
            after: roleState.lastEntrant.delayedApplause
        },
        directorCueTokens: {
            before: game.roleState.curtainDirector.cueTokens,
            after: roleState.curtainDirector.cueTokens
        },
        witnessedHandTypesAdded: witnessedAdded
    };
    const timelineEntry = {
        scene: game.sceneNumber,
        natural: naturalEvaluation.type,
        final: finalEvaluation.type,
        modifier: finalEvaluation.modificationLabel,
        scoreItems: scoreItems.map((item) => ({...item})),
        totalScore,
        triggeredRoleIds: [...triggeredRoleIds],
        effectLog: effectLog.map((entry) => ({...entry}))
    };
    const resolution = {
        valid: true,
        errorCode: null,
        errorMessage: null,
        naturalEvaluation,
        finalEvaluation,
        scoreItems,
        totalScore,
        varietyBefore,
        varietyAfter,
        repeated,
        triggeredRoleIds,
        effectLog,
        resourceChanges,
        timelineEntry
    };
    return {
        resolution,
        commit: {
            actScoreDelta: totalScore,
            runScoreDelta: totalScore,
            previousHandType: finalEvaluation.type,
            varietyLevel: varietyAfter,
            rehearsals,
            witnessedHandTypes,
            roleState,
            timelineEntry
        }
    };
};

const refillAfterRemoval = function (state, removedCards) {
    let drawPile = [...state.drawPile];
    let discardPile = [...state.discardPile];
    let rngState = state.rngState;
    const drawn = [];
    while (drawn.length < removedCards.length) {
        if (drawPile.length === 0) {
            if (discardPile.length === 0) {
                throw new Error("Deck invariant violated: no cards available.");
            }
            const result = shuffleWithSeed(rngState, discardPile);
            rngState = result.rngState;
            drawPile = result.shuffled;
            discardPile = [];
        }
        if (drawPile.length === 0) {
            throw new Error(
                "Deck invariant violated: shuffle returned no cards."
            );
        }
        drawn.push(drawPile[0]);
        drawPile = drawPile.slice(1);
    }
    return {
        drawn,
        drawPile,
        discardPile: [...discardPile, ...removedCards],
        rngState
    };
};

const createRewindSnapshot = (game) => ({
    rngState: game.rngState,
    actScore: game.actScore,
    runScore: game.runScore,
    sceneNumber: game.sceneNumber,
    rehearsals: game.rehearsals,
    varietyLevel: game.varietyLevel,
    previousHandType: game.previousHandType,
    timeline: game.timeline,
    bannedHandTypes: game.bannedHandTypes,
    lastSceneResult: game.lastSceneResult,
    witnessedHandTypes: game.witnessedHandTypes,
    drawPile: game.drawPile,
    hand: game.hand,
    discardPile: game.discardPile,
    selectedCardIds: game.selectedCardIds,
    roleState: game.roleState,
    lastError: game.lastError
});

const advance = function (game) {
    if (game.sceneNumber < 5) {
        return {
            ...game,
            phase: "playing",
            sceneNumber: game.sceneNumber + 1,
            pendingScene: null,
            lastError: null
        };
    }
    const target = ACT_TARGETS[game.currentAct - 1];
    if (game.actScore < target) {
        return {...game, phase: "lost", pendingScene: null, lastError: null};
    }
    if (game.currentAct === 3) {
        return {...game, phase: "won", pendingScene: null, lastError: null};
    }
    return {
        ...game,
        phase: "act-complete",
        pendingScene: null,
        lastError: null
    };
};

const applySceneCommit = function (game, resolution, commit, snapshotBefore) {
    const removedCards = selectedCards(game);
    const removedIds = removedCards.map((card) => card.id);
    const handAfterRemoval = game.hand.filter(
        (card) => !removedIds.includes(card.id)
    );
    const refill = refillAfterRemoval(game, removedCards);
    const candidate = {
        ...game,
        rngState: refill.rngState,
        actScore: game.actScore + commit.actScoreDelta,
        runScore: game.runScore + commit.runScoreDelta,
        rehearsals: commit.rehearsals,
        varietyLevel: commit.varietyLevel,
        previousHandType: commit.previousHandType,
        timeline: [...game.timeline, commit.timelineEntry],
        witnessedHandTypes: commit.witnessedHandTypes,
        hand: [...handAfterRemoval, ...refill.drawn],
        drawPile: refill.drawPile,
        discardPile: refill.discardPile,
        selectedCardIds: [],
        roleState: commit.roleState,
        lastSceneResult: resolution,
        lastError: null
    };
    const shouldPend = (
        game.castSlots.includes("marionette") &&
        !game.roleState.marionette.rewindUsed
    );
    if (shouldPend) {
        return {
            ...candidate,
            phase: "scene-pending",
            pendingScene: {snapshotBefore, resolution}
        };
    }
    return advance(candidate);
};

/**
 * Create a reproducible title-state game without dealing cards.
 * @memberof LastCurtain
 * @param {number|string} seed Reproducible seed.
 * @returns {GameState} New title-state game.
 */
export function createGame(seed) {
    if (typeof seed !== "number" && typeof seed !== "string") {
        throw new TypeError("A game seed must be a number or string.");
    }
    const roles = shuffleWithSeed(seedToState(seed), INITIAL_ROLE_ORDER);
    return {
        initialSeed: seed,
        rngState: roles.rngState,
        phase: "title",
        auditionType: null,
        offeredRoleIds: [],
        auditionQueue: roles.shuffled,
        castSlots: [],
        currentAct: 1,
        actScore: 0,
        runScore: 0,
        sceneNumber: 1,
        rehearsals: 2,
        varietyLevel: 0,
        previousHandType: null,
        timeline: [],
        bannedHandTypes: [],
        pendingScene: null,
        lastSceneResult: null,
        witnessedHandTypes: [],
        drawPile: [],
        hand: [],
        discardPile: [],
        selectedCardIds: [],
        roleState: initialRoleState(),
        lastError: null
    };
}

/**
 * Open the first audition without dealing the act deck.
 * @memberof LastCurtain
 * @param {GameState} game Title-state game.
 * @returns {GameState} Opening audition or error state.
 */
export function startGame(game) {
    if (game.phase !== "title") {
        return withError(game, "WRONG_PHASE");
    }
    return {
        ...game,
        phase: "audition",
        auditionType: "opening",
        offeredRoleIds: game.auditionQueue.slice(0, 3),
        lastError: null
    };
}

const resolveAuditionQueue = function (queue, chosenId) {
    const offered = queue.slice(0, 3);
    const untouched = queue.slice(3);
    return [
        ...untouched,
        ...offered.filter((id) => id !== chosenId)
    ];
};

const initialiseAct = function (game, rngState, auditionQueue, castSlots) {
    const shuffledDeck = shuffleWithSeed(rngState, createDeck());
    return {
        ...game,
        rngState: shuffledDeck.rngState,
        phase: "playing",
        auditionType: null,
        offeredRoleIds: [],
        auditionQueue,
        castSlots,
        actScore: 0,
        sceneNumber: 1,
        rehearsals: 2,
        varietyLevel: 0,
        previousHandType: null,
        timeline: [],
        bannedHandTypes: [],
        pendingScene: null,
        lastSceneResult: null,
        drawPile: shuffledDeck.shuffled.slice(8),
        hand: shuffledDeck.shuffled.slice(0, 8),
        discardPile: [],
        selectedCardIds: [],
        roleState: initialRoleState(),
        lastError: null
    };
};

/**
 * Recruit an offered role and begin the corresponding act.
 * @memberof LastCurtain
 * @param {GameState} game Audition-state game.
 * @param {string} roleId Offered role identifier.
 * @returns {GameState} Playing game or player-facing error state.
 */
export function chooseRole(game, roleId) {
    if (game.phase !== "audition") {
        return withError(game, "WRONG_PHASE");
    }
    if (game.castSlots.includes(roleId)) {
        return withError(game, "ROLE_ALREADY_RECRUITED");
    }
    if (game.castSlots.length >= 3) {
        return withError(game, "CAST_FULL");
    }
    if (!game.offeredRoleIds.includes(roleId)) {
        return withError(game, "INVALID_CHOICE");
    }
    let queue = resolveAuditionQueue(game.auditionQueue, roleId);
    let rngState = game.rngState;
    if (game.auditionType === "opening") {
        const insertion = randomInt(rngState, 0, queue.length + 1);
        rngState = insertion.rngState;
        queue = [
            ...queue.slice(0, insertion.value),
            "curtainDirector",
            ...queue.slice(insertion.value)
        ];
    }
    const currentAct = (
        game.auditionType === "opening"
        ? 1
        : game.currentAct + 1
    );
    return initialiseAct(
        {...game, currentAct},
        rngState,
        queue,
        [...game.castSlots, roleId]
    );
}

/** Return the offered role definitions in audition order. */
export function getAuditionCandidates(game) {
    if (game.phase !== "audition") {
        return [];
    }
    return game.offeredRoleIds.map((id) => ROLE_DEFINITIONS[id]);
}

const roleStatus = function (game, roleId, slot) {
    if (roleId === "lastEntrant") {
        return `DELAYED: ${game.roleState.lastEntrant.delayedApplause}`;
    }
    if (roleId === "faceless") {
        return game.roleState.faceless.ready
            ? "READY"
            : "EXHAUSTED — CHANGE THE HAND TO RESTORE";
    }
    if (roleId === "witnesses") {
        return `WITNESSED: ${game.witnessedHandTypes.length} / 9`;
    }
    if (roleId === "twinMask") {
        return game.roleState.twinMask.face === "comedy"
            ? "COMEDY — RED"
            : "TRAGEDY — BLACK";
    }
    if (roleId === "curtainDirector") {
        return slot === 0
            ? "DORMANT — NO ROLE TO THE LEFT"
            : `CUES: ${game.roleState.curtainDirector.cueTokens} / 2`;
    }
    return game.roleState.marionette.rewindUsed
        ? "RETAKE USED"
        : "RETAKE AVAILABLE";
};

/** Return display-ready views of the current cast without exposing UI rules. */
export function getCastRoleViews(game) {
    return game.castSlots.map(function (id, slot) {
        return {
            ...ROLE_DEFINITIONS[id],
            slot,
            status: roleStatus(game, id, slot),
            dormant: id === "curtainDirector" && slot === 0
        };
    });
}

/** Return cards the player may currently toggle. */
export function getSelectableCardIds(game) {
    if (game.phase !== "playing") {
        return [];
    }
    return game.selectedCardIds.length >= 5
        ? [...game.selectedCardIds]
        : game.hand.map((card) => card.id);
}

/** Return legal stage directions for one recruited role. */
export function getMovableDirections(game, roleId) {
    if (game.phase !== "playing") {
        return [];
    }
    const index = game.castSlots.indexOf(roleId);
    if (index < 0) {
        return [];
    }
    return [
        ...(index > 0 ? ["left"] : []),
        ...(index < game.castSlots.length - 1 ? ["right"] : [])
    ];
}

/** Swap a recruited role with an adjacent stage slot. */
export function moveRole(game, roleId, direction) {
    if (game.phase !== "playing") {
        return withError(game, "WRONG_PHASE");
    }
    const index = game.castSlots.indexOf(roleId);
    if (index < 0) {
        return withError(game, "ROLE_NOT_IN_CAST");
    }
    if (direction !== "left" && direction !== "right") {
        return withError(game, "INVALID_DIRECTION");
    }
    const destination = index + (direction === "left" ? -1 : 1);
    if (destination < 0 || destination >= game.castSlots.length) {
        return withError(game, "MOVE_OUT_OF_BOUNDS");
    }
    const castSlots = [...game.castSlots];
    castSlots[index] = castSlots[destination];
    castSlots[destination] = roleId;
    return {...game, castSlots, lastError: null};
}

/** Toggle one card in the proposed scene, up to five selected cards. */
export function toggleCardSelection(game, cardId) {
    if (game.phase !== "playing") {
        return withError(game, "WRONG_PHASE");
    }
    if (!game.hand.some((card) => card.id === cardId)) {
        return withError(game, "CARD_NOT_IN_HAND");
    }
    if (game.selectedCardIds.includes(cardId)) {
        return {
            ...game,
            selectedCardIds: game.selectedCardIds.filter((id) => id !== cardId),
            lastError: null
        };
    }
    if (game.selectedCardIds.length >= 5) {
        return withError(game, "TOO_MANY_CARDS");
    }
    return {
        ...game,
        selectedCardIds: [...game.selectedCardIds, cardId],
        lastError: null
    };
}

/** Return a deterministic scene preview without committing it. */
export function previewScene(game) {
    return resolveScene(game).resolution;
}

/** Commit the selected cards as a scene, possibly opening a retake decision. */
export function performScene(game) {
    const result = resolveScene(game);
    if (!result.resolution.valid || !result.commit) {
        return withError(
            game,
            result.resolution.errorCode,
            result.resolution.errorMessage
        );
    }
    return applySceneCommit(
        game,
        result.resolution,
        result.commit,
        createRewindSnapshot(game)
    );
}

/** Keep a pending performed scene and advance the lifecycle. */
export function confirmScene(game) {
    if (game.phase !== "scene-pending") {
        return withError(game, "WRONG_PHASE");
    }
    return advance({...game, pendingScene: null, lastError: null});
}

/** Restore the state before a pending scene and ban its final hand. */
export function rewindScene(game) {
    if (game.phase !== "scene-pending") {
        return withError(game, "WRONG_PHASE");
    }
    if (!game.pendingScene) {
        throw new Error("Pending-scene invariant violated.");
    }
    const snapshot = game.pendingScene.snapshotBefore;
    const finalType = game.pendingScene.resolution.finalEvaluation.type;
    return {
        ...game,
        ...snapshot,
        phase: "playing",
        pendingScene: null,
        bannedHandTypes: addUnique(snapshot.bannedHandTypes, finalType),
        roleState: {
            ...copyRoleState(snapshot.roleState),
            marionette: {rewindUsed: true}
        },
        lastError: null
    };
}

/** Replace the selected cards without allowing them to be drawn immediately. */
export function rehearse(game) {
    if (game.phase !== "playing") {
        return withError(game, "WRONG_PHASE");
    }
    if (game.rehearsals <= 0) {
        return withError(game, "NO_REHEARSALS_LEFT");
    }
    const errorCode = selectionError(game);
    if (errorCode) {
        return withError(game, errorCode);
    }
    const removedCards = selectedCards(game);
    const removedIds = removedCards.map((card) => card.id);
    const refill = refillAfterRemoval(game, removedCards);
    const roleState = copyRoleState(game.roleState);
    if (game.castSlots.includes("twinMask")) {
        roleState.twinMask.face = (
            roleState.twinMask.face === "comedy"
            ? "tragedy"
            : "comedy"
        );
    }
    return {
        ...game,
        rngState: refill.rngState,
        rehearsals: game.rehearsals - 1,
        hand: [
            ...game.hand.filter((card) => !removedIds.includes(card.id)),
            ...refill.drawn
        ],
        drawPile: refill.drawPile,
        discardPile: refill.discardPile,
        selectedCardIds: [],
        roleState,
        lastError: null
    };
}

/** Move from a completed act to its intermission audition. */
export function continueAfterAct(game) {
    if (game.phase !== "act-complete") {
        return withError(game, "WRONG_PHASE");
    }
    return {
        ...game,
        phase: "audition",
        auditionType: game.currentAct === 1 ? "post-1" : "post-2",
        offeredRoleIds: game.auditionQueue.slice(0, 3),
        lastError: null
    };
}

/** Fully recreate a title-state game, preserving the original seed by default. */
export function restartGame(game, seed = game.initialSeed) {
    return createGame(seed);
}

/** Return the current lifecycle phase. */
export function getPhase(game) {
    return game.phase;
}

/** Return the applause target derived from the current act. */
export function getActTarget(game) {
    return ACT_TARGETS[game.currentAct - 1];
}

/** Return phase and state-aware action identifiers for the UI. */
export function getAvailableActions(game) {
    if (game.phase === "title") {
        return ["start"];
    }
    if (game.phase === "audition") {
        return (
            game.offeredRoleIds.length > 0 && game.castSlots.length < 3
            ? ["choose-role"]
            : []
        );
    }
    if (game.phase === "playing") {
        const actions = game.hand.length > 0 ? ["toggle-card"] : [];
        if (previewScene(game).valid) {
            actions.push("perform");
        }
        if (
            game.rehearsals > 0 &&
            selectionError(game) === null
        ) {
            actions.push("rehearse");
        }
        if (game.castSlots.length > 1) {
            actions.push("move-role");
        }
        return actions;
    }
    if (game.phase === "scene-pending") {
        return ["confirm", "rewind"];
    }
    if (game.phase === "act-complete") {
        return ["continue"];
    }
    return ["restart"];
}

/** Return whether the run has reached victory or defeat. */
export function isGameOver(game) {
    return game.phase === "won" || game.phase === "lost";
}
